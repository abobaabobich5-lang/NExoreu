import os
import random
import logging
from telegram import Update
from telegram.ext import Application, MessageHandler, filters, ContextTypes

# Включаем логирование
logging.basicConfig(
    format="%(asctime)s - %(name)s - %(levelname)s - %(message)s", 
    level=logging.INFO
)

# Токен бота (ЗАМЕНИТЕ НА СВОЙ!)
BOT_TOKEN = "7750901927:AAHh-0lcH77bHvblFgIrs_W6-7WDk-U4EO8"

# Путь к папке с карточками
CARDS_FOLDER = "card"

def get_card_files():
    """Получает список всех картинок в папке"""
    try:
        files = [f for f in os.listdir(CARDS_FOLDER) 
                if f.lower().endswith(('.png', '.jpg', '.jpeg'))]
        return files
    except:
        return []

async def handle_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Обрабатывает сообщения"""
    # Проверяем, есть ли слово "карточка"
    if "карточка" in update.message.text.lower():
        cards = get_card_files()
        
        if not cards:
            await update.message.reply_text("😕 Карточки закончились!")
            return
        
        # Выбираем случайную карточку
        chosen = random.choice(cards)
        card_path = os.path.join(CARDS_FOLDER, chosen)
        
        # Отправляем
        with open(card_path, 'rb') as photo:
            await update.message.reply_photo(
                photo=photo,
                caption=f"🎴 Тебе выпала: {chosen}"
            )

def main():
    # Создаем бота
    app = Application.builder().token(BOT_TOKEN).build()
    
    # Добавляем обработчик
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, handle_message))
    
    # Запускаем
    print("Бот запущен!")
    app.run_polling()

if __name__ == "__main__":

    main()
